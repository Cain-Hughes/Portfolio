# proxmox_manager.py
import proxmoxer
from datetime import datetime

class ProxmoxLab:
    def __init__(self, host, user, passwd):
        self.api = proxmoxer.ProxmoxAPI(
            host, user=user,
            password=passwd,
            verify_ssl=False
        )
        self.node = 'pve-homelab'

    def list_vms(self):
        vms = self.api.nodes(self.node).qemu.get()
        for vm in sorted(vms, key=lambda x: x['vmid']):
            name = vm['name']
            status = vm['status']
            print(f"  {vm['vmid']} | {name:20s} | {status}")

    def snapshot_all(self, label='auto-backup'):
        for vm in self.api.nodes(self.node).qemu.get():
            if vm['status'] == 'running':
                self.api.nodes(self.node) \
                    .qemu(vm['vmid']) \
                    .snapshot.create(snapname=label)
                print(f"  [+] Snapshot: {vm['name']}")

    def create_vm(self, vmid, name, cores, mem):
        self.api.nodes(self.node).qemu.create(
            vmid=vmid, name=name,
            cores=cores, memory=mem,
            net0='virtio,bridge=vmbr0',
            scsi0='local-lvm:32',
            ostype='l26'
        )

import requests

class AdGuardDNS:
    def __init__(self, url, auth):
        self.url = url
        self.auth = auth

    def stats(self):
        r = requests.get(
            f"{self.url}/control/stats",
            auth=self.auth
        )
        data = r.json()
        print(f"  Queries:  {data['num_dns_queries']}")
        print(f"  Blocked:  {data['num_blocked_filtering']}")

    def add_rewrite(self, domain, ip):
        requests.post(
            f"{self.url}/control/rewrite/add",
            auth=self.auth,
            json={"domain": domain, "answer": ip}
        )
